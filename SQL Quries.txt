create table employee
(

  age int
  
);

create table product
(
  id int primary key,
  name varchar(50) unique,
  description varchar(300) 
);

CREATE TABLE orders
(
  id INT PRIMARY KEY,
  product_id INT,
  order_date DATE,
  -- Additional columns for the orders table
  FOREIGN KEY (product_id) REFERENCES product(id)
);
